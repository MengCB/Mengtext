2. 排序补充： 
冒泡排序算法
var a = [3,5,7,8,9,33,44,55];
for( var i=0; i<a.length-1 ; i++){
	for( var j=0; j<a.length-1-i; j++ ){
		if( a[j] > a[j+1] ){
			var temp = a[j];
			a[j] = a[j+1];
			a[j+1] = temp;
		}
	}
}
console.info(a);
